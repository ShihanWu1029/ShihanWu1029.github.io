放苹果

发表于 2023-08-10 分类于 题解

AC 代码

#include <bits/stdc++.h>
using namespace std;
int f(int m,int n)
{
	if(m==0||n==1) 
	{
		return 1;	
	}
	if(n>m)
	{
		return f(m,m);	
	}
	else
	{
		return f(m,n-1)+f(m-n,n);
	}
}
int main()
{
	int m,n,c;
	cin>>c;
	while(c--)
	{
		cin>>m>>n;
		cout<<f(m,n);
	}
	return 0;
}
