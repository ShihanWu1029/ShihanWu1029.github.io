AC 代码

#include<iostream>
#include<cstring>
using namespace std;
int pow[17];
void print(int k){
    int t=15;
    int a[17];
    int len=0;
    memset(a,0,sizeof(a));
    while(k>0){
        if(k>=pow[t]){
            a[++len]=t;
            k-=pow[t];
        }
        t--;
    }
    for(int i=1;i<=len;i++){
        if(a[i]==0)
            cout<<"2(0)";
        else if(a[i]==1)
            cout<<"2";
        else{
            cout<<"2(";
            print(a[i]);
            cout<<")";
        }
        if(i!=len)
            cout<<'+';
    }
    return ;
}
int main(){
    ios::sync_with_stdio(false);
    pow[0]=1;
    for(int i=1;i<=15;i++)
        pow[i]=2*pow[i-1];
    int n;
    cin>>n;
    print(n);
    cout<<endl;
    return 0;
}
