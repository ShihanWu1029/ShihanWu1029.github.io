珠心算测验

发表于 2023-08-09 更新于 2023-08-10 分类于 题解

AC 代码

#include <iostream>
using namespace std;

int a[10000];
int main(){
    int n,nums[1001];
    cin>>n;
    for (int i=0;i<n;++i){
        int t;
        cin>>t;
        nums[i] = t;
    }
    for (int i=0;i<n;i++) {
        for (int j=i+1;j<n;j++) {
            for (int k=0;k<n;k++) {
                if (nums[i]+nums[j] == nums[k]){
                    a[nums[k]]+=1;
                }
            }
        }
    }
    int cnt = 0;
    for (int i=0;i<10000;i++){
        if (a[i] != 0){
            cnt++;
        }
    }
    cout<<cnt<<endl;
    return 0;
}
