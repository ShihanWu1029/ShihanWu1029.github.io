幂次方

AC 代码

#include<iostream>
#include<cstring>
using namespace std;
int pow[17];
void print(int k){
    int t=15;
    int a[17];
    int len=0;
    memset(a,0,sizeof(a));
    while(k>0){
        if(k>=pow[t]){
            a[++len]=t;
            k-=pow[t];
        }
        t--;
    }
    for(int i=1;i<=len;i++){
        if(a[i]==0)
            cout<<"2(0)";
        else if(a[i]==1)
            cout<<"2";
        else{
            cout<<"2(";
            print(a[i]);
            cout<<")";
        }
        if(i!=len)
            cout<<'+';
    }
    return ;
}
int main(){
    ios::sync_with_stdio(false);
    pow[0]=1;
    for(int i=1;i<=15;i++)
        pow[i]=2*pow[i-1];
    int n;
    cin>>n;
    print(n);
    cout<<endl;
    return 0;
}

小凯的烦恼

发表于 2023-08-10 分类于 题解

AC代码

数学方法

#include<iostream>
using namespace std;
int main(){
	long long a,b;
	cin>>a>>b;
	cout<<a*b-a-b<<endl;
	return 0;
}

证明

我们不妨设 a<b，答案为x

如果x可以被a和b表示出来的话，那么就有

x=ma+nb(m≥0,n≥0)

但是x不能表达成上面的式子，我们要使x最大，因为a<b，所以令n=−1

而且ma不能被b整除，否则b又会多出一个因子

因此m_max=b−1所以x=a(b−1)−b=ab−a−b

即使当a>b时推出的式子完全相同，因此最终的答案为ab−a−b

    for (int i = 0; i < 6; ++ i) {
        A[i][0] = INIT[i][0] * INIT[i][0];
        A[i][1] = INIT[i][1] * INIT[i][1];
        A[i][2] = INIT[i][0] * INIT[i][1];
        A[i][3] = INIT[i][0];
        A[i][4] = INIT[i][1];
        A[i][5] = 1;
        A[i][6] = INIT[i][2];
    }
    for (int i = 0; i < 6; ++ i) {
        for (int j = i + 1; j < 6; ++ j) {
            if (abs(A[j][i]) > abs(A[i][i])) {
                for (int k = 0; k < 7; ++ k) swap(A[i][k], A[j][k]);
            }
        }
        for (int j = 6; j >= i; -- j)
            A[i][j] /= A[i][i];
        for (int j = i + 1; j < 6; ++ j) {
            for (int k = 6; k >= i; --k) {
                A[j][k] -= A[j][i] * A[i][k];
            }
        }
    }
    for (int i = 5; ~i; -- i) {
        for (int j = i - 1; ~j; -- j) {
            A[j][6] -= A[j][i] * A[i][6];
            A[j][i] = 0;
        }
    }
}

3.例题： 大小写转换

题目描述

给定一个字符串，将其中的大写字母转换成小写，小写转换成大写。

输入样例：

We love C++!

错误示范

int main() {
    string s;
    getline(cin,s);
    for(int i=0;i<s.size();i++){
        if(islower(s[i])) s[i]-=32;
        if(isupper(s[i])) s[i]+=32;
    }
    cout<<s;
}

如果用上面所给的代码，会发现只有大写变成了小写，儿小写没有变成大写。

问题出在：

if(islower(s[i])) s[i]-=32;
if(isupper(s[i])) s[i]+=32;

两个if函数并列使用但是没有else，因此在小写变成大写之后又被变回了小写。

应改为：

if(islower(s[i])) s[i]-=32;
else if(isupper(s[i])) s[i]+=32;

时空复杂度

定义：一个可执行文件在时间和空间上的效率

数据结构和算法要解决的问题就是让代码运行更快。

一般来说，我们会以空间换时间

时间复杂度的分析

表达的是一种趋势，而非绝对意义上的时间。

分析时间复杂度时，我们会忽略输入与输出所耗费的时间与常数和系数。

如果一段代码只有常数级时间复杂度，则写为O(1)。

举例

如图是一个求1~N的累加和

int function(int n){
    int sum=0;
    for(int i=0;i<=n;i++){
        sum+=i;
    }
    return sum;
}

这一段代码的时间复杂度应该如何去分析呢？

首先我们易知3，4行代码会执行n次，然后包含有for的执行条件与sum的累加，因此算为2n。然后又有int语句与return，因此时间复杂度为O(2+2n)。

但是我们要忽略常数和系数，因此最终答案为O(n)。

常见时间复杂度

  • O(1)：常数级
  • O(log n)：对数级
  • O(n)：线性级
  • O(n log n)：线性对数级
  • O(n^2)：平方级
  • O(n^3)：立方级
  • O(2^n)：指数级
  • O(n!)：阶乘级

空间复杂度的分析

空间复杂度是对一个算法在运行过程中临时占用储存空间大小的量度。

空间复杂度不是指程序本身占用的空间，而是指诸如数组之类变量所占的空间。

例题

vector<int> function(int n){
    vector<int> v;
    for(int i=1;i<=n;i++){
        v.push_back(i);
    }
    return v;
}

如上我们在函数内定义了一个vector容器，然后往里面push了n个值，因此该函数的空间复杂度为O(n)。

由此可以归纳分析空间复杂度时应考虑一个变量或容器在运行时所占的最大空间。

递归的时间复杂度分析

int fun(int n){
    if(n==1)return 1;
    return fun(n-1)+n;
}

由于C++在递归过程中会调用栈内存，因此我们也需要考虑空间复杂度。

如上是一个简单的递归，每一次都会向函数本身请求fun(n-1)直到n=1，因此调用了n次栈，故空间复杂度为O(n)。

Spell Checker

发表于 2023-08-09 更新于 2023-08-10 分类于 题解

    #include <iostream>
#include <map>
#include <set>
#include <list>
#include <string>

using namespace std;

map<string, int> dictionary;
set<size_t> lens;
bool comp(string a, string b) {
    return (dictionary[a] - dictionary[b]) < 0 ? true : false;
}

int main() {
    string buff;
    int index = 0;
    while (cin >> buff && buff != "#") {
        dictionary[buff] = ++index;
        size_t length = buff.size();
        lens.insert(length);
    }
    while (cin >> buff && buff != "#") {
        size_t length = buff.size();
        list<string> queue;
        if (lens.find(length) != lens.end()) {
            if (dictionary.find(buff) != dictionary.end()) {
                cout << buff << " is correct" << endl;
                continue;
            }
            for (char c = 'a'; c <= 'z'; ++c) {
                for (int i = 0; i < buff.size(); ++i) {
                    string temp = buff;
                    temp.replace(i, 1, 1, c);
                    if (dictionary.find(temp) != dictionary.end()) {
                        queue.push_back(temp);
                    }
                }
            }
        }
        if (lens.find(length - 1) != lens.end()) {
            for (int i = 0; i < buff.size(); ++i) {
                string temp = buff;
                temp.erase(i, 1);
                if (dictionary.find(temp) != dictionary.end()) {
                    queue.push_back(temp);
                }
            }
        }
        if (lens.find(length + 1) != lens.end()) {
            for (char c = 'a'; c <= 'z'; ++c) {
                for (int i = 0; i <= buff.size(); ++i) {
                    string temp = buff;
                    temp.insert(i, 1, c);
                    if (dictionary.find(temp) != dictionary.end()) {
                        queue.push_back(temp);
                    }
                }
            }
        }
        queue.unique();
        queue.sort(comp);
        cout << buff << ":";
        for (list<string>::iterator it = queue.begin(); it != queue.end(); ++it) {
            cout << " " << *it;
        }
        cout << endl;
    }
    return 0;
}

比例简化

#include<iostream>
#include<algorithm>
using namespace std;
int a,b,l,ansa,ansb;
int main(){
	cin>>a>>b>>l;
	ansa=l;
    ansb=1;
	for(int i=1;i<=l;i++) {
		for (int j=1;j<=l;j++) {
			if (__gcd(i,j) == 1 && i*b >= j*a && ansa*j > i*ansb)
				ansa=i,ansb=j;
		}
	}
	cout<<ansa<<" "<<ansb<<endl;
	return 0;
}

/*
i/j>=a/b <=> i*b>=j*a
ansa/ansb > i/j  <=> ansa*j>ansb*i
*/

珠心算测验

#include <iostream>
using namespace std;

int a[10000];
int main(){
    int n,nums[1001];
    cin>>n;
    for (int i=0;i<n;++i){
        int t;
        cin>>t;
        nums[i] = t;
    }
    for (int i=0;i<n;i++) {
        for (int j=i+1;j<n;j++) {
            for (int k=0;k<n;k++) {
                if (nums[i]+nums[j] == nums[k]){
                    a[nums[k]]+=1;
                }
            }
        }
    }
    int cnt = 0;
    for (int i=0;i<10000;i++){
        if (a[i] != 0){
            cnt++;
        }
    }
    cout<<cnt<<endl;
    return 0;
}
